2   package pasianssi.ui;
3   
4   import java.awt.event.ActionEvent;
5   import java.awt.event.ActionListener;
6   import java.util.ArrayList;
7   import java.util.HashSet;
8   import pasianssi.util.TilanteenTarkastaja;
9   /**
10   * @author Johanna
11   */
12  public class VihjeNapinKuuntelija implements ActionListener {
13      private TilanteenTarkastaja tt;
14      private int kerrat;
15      private boolean paalla;
16      private ArrayList<PaikkaLabel> vihjeena;
17      
18      public VihjeNapinKuuntelija(TilanteenTarkastaja tt) {
19          this.tt = tt;
20          this.kerrat = 1;
21          this.paalla = false;
22      }
23      
24      public void nollaaKerrat() {
25          this.kerrat = 0;
26      }
27      
28      public boolean onkoPaalla() {
29          return this.paalla;
30      }
31      
32      public void poisPaalta() {
33          this.paalla = false;
34          this.kerrat = 1;
35          this.vihjeena.clear();
36      }
37      
38      public ArrayList<PaikkaLabel> getLista() {
39          return this.vihjeena;
40      }
41      
42      public void pois() {
43          if (this.vihjeena.isEmpty()) {
44              poisPaalta();
45          } else if (this.vihjeena.size() == 2) {
46              this.vihjeena.get(0).getPaikka().poistaValinta();
47              this.vihjeena.get(1).getPaikka().poistaValinta();
48              this.vihjeena.get(0).repaint();
49              this.vihjeena.get(1).repaint();
50              this.vihjeena.clear();
51          } else {
52              this.vihjeena.get(0).getPaikka().poistaValinta();
53              this.vihjeena.get(1).getPaikka().poistaValinta();
54              this.vihjeena.get(2).getPaikka().poistaValinta();
55              this.vihjeena.get(0).repaint();
56              this.vihjeena.get(1).repaint();
57              this.vihjeena.get(2).repaint();
58              this.vihjeena.clear();
59          }
60      }
61  
62      @Override
63      public void actionPerformed(ActionEvent e) {
64          if (onkoPaalla()) {
65              pois();
66          }
67          this.vihjeena = new ArrayList<>();
68          HashSet<PaikkaLabel[]> parit = tt.haeParit();
69          PaikkaLabel[] paikkaset = null;
70          if (parit != null && !parit.isEmpty()) {
71              this.paalla = true;
72              int parimaara = 0;
73              for (PaikkaLabel[] paikka : parit) {
74                  parimaara++;
75              }
76              if (this.kerrat > parimaara) {
77                  this.kerrat = 1;
78              }
79              int i = 1;
80              for (PaikkaLabel[] paikat:parit) {
81                  if (i == this.kerrat) {
82                      paikkaset = paikat;
83                  }
84                  i++;
85              }
86              PaikkaLabel paikka1 = paikkaset[0];
87              PaikkaLabel paikka2 = paikkaset[1];
88              this.vihjeena.add(paikka1);
89              this.vihjeena.add(paikka2);
90              paikka1.getPaikka().asetaValituksi();
91              paikka2.getPaikka().asetaValituksi();
92              paikka1.repaint();
93              paikka2.repaint();
94              if (paikkaset.length == 3) {
95                  PaikkaLabel paikka3 = paikkaset[2];
96                  this.vihjeena.add(paikka3);
97                  paikka3.getPaikka().asetaValituksi();
98                  paikka3.repaint();
99              }
100             this.kerrat++;
101         }
102     }
103 }
