4   import java.awt.event.ActionEvent;
5   import java.awt.event.ActionListener;
6   import java.io.IOException;
7   import java.util.logging.Level;
8   import java.util.logging.Logger;
9   import javax.swing.JOptionPane;
10  /**
11   * @author Johanna
12   */
13  public class UusiPeliNapinkuuntelija implements ActionListener {
14      private Kayttoliittyma kl;
15      
16      public UusiPeliNapinkuuntelija(Kayttoliittyma kl) {
17          this.kl = kl;
18      }
19      
20      @Override
21      public void actionPerformed(ActionEvent e) {
22          int valinta = JOptionPane.showConfirmDialog(null, "Haluatko varmasti aloittaa uuden pelin?");
23          if (valinta == JOptionPane.YES_OPTION) {
24              try {
25                  kl.reset(kl.getFrame().getContentPane());
26              } catch (IOException ex) {
27                  Logger.getLogger(PaikkaLabelHiirenKuuntelija.class.getName()).log(Level.SEVERE, null, ex);
28              }
29          }
30      }
31  }
