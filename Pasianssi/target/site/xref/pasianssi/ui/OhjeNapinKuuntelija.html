4   import java.awt.event.ActionEvent;

7   /**

10  public class OhjeNapinKuuntelija implements ActionListener {
11      private VihjeNapinKuuntelija vnk;
12      
13      public OhjeNapinKuuntelija(VihjeNapinKuuntelija vnk) {
14          this.vnk = vnk;
15      }
16  
17      @Override
18      public void actionPerformed(ActionEvent e) {
19          if (this.vnk.onkoPaalla()) {
20              this.vnk.pois();
21          }
22          JOptionPane.showMessageDialog(null, "Pelissä tarkoituksena on tyhjentää pakka ja pöytä korteista.\n"
23                                              + "Kortit voidaan poistaa pareina, jos niiden summa on 11 ja \n"
24                                              + "ne sijaitsevat samalla pysty- tai vaakarivillä. Kuvakortit \n"
25                                              + "voidaan poistaa samalta pysty- tai vaakariviltä, jos niiden \n"
26                                              + "välissä ei ole muita kortteja ja ne ovat järjestyksessä J, Q, K \n"
27                                              + "tai K, Q, J. Kortit voidaan valita missä järjestyksessä tahansa, \n"
28                                              + "kunhan niiden järjestys vain pöydällä on oikea. \n"
29                                              + "Peli aloitetaan painamalla pakkaa, jolloin kortit jaetaan pöydälle. \n"
30                                              + "Aika kuitenkin lasketaan jo siitä, kun uusi peli aloitetaan, vaikka \n"
31                                              + "kortteja ei ole vielä jaettu pöydälle. \n"
32                                              + "Kun pareja tai järjestyksessä olevia kuvakortteja ei enää löydy \n"
33                                              + "pöydältä, jaetaan pakasta lisää kortteja klikkaamalla pakkaa.\n"
34                                              + "Kun pakka on tyhjä, jaetaan kortit pöydän tyhjiin paikkoihin klikkaamalla \n"
35                                              + "pakkaa. Kortit siirtyvät niin, että viimeisenä oleva kortti pöydällä \n"
36                                              + "siirretään aina ensimmäiseen tyhjään paikkaan pöydällä. \n"
37                                              + "Jos ei löydä pareja tai poistettavia kuvakortteja, painamalla \"Vinkki\"-nappia \n"
38                                              + "näkee mahdolliset poistettavat korttiparit/kuvakorttiryhmät. Jos ei nappia \n"
39                                              + "painammalla tapahdu mitään, ainut mahdollisuus on klikata pakkaa ja jakaa \n"
40                                              + "lisää kortteja/siirtää pöydällä olevia kortteja. \n"
41                                              + "Jos mahdollisia poistoja ei enää ole, eikä pakkaa voi käyttää, peli päättyy ja \n"
42                                              + "siitä ilmoitetaan pelaajalle. \n"
43                                              + "Peli on mennyt läpi, kun pakka sekä pöytä ovat tyhjät korteista.");
44      }
