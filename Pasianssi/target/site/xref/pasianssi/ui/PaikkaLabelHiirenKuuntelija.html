52      public int korttienMaara() {
53          int pakanKortit = pakka.pakanKortit().size();
54          int poydanTyhjat = 25 - poyta.getTyhjat().size();
55          return pakanKortit + poydanTyhjat;
56      }
57      
58      public String kaytettyAika() {
59          int s = tk.getKokonaisSekunnit();
60          int tunnit = s / 360;
61          int minuutit = s / 60;
62          if (minuutit > 59) {
63              minuutit = minuutit % 60;
64          }
65          int sekunnit = s;
66          if (sekunnit > 59) {
67              sekunnit = sekunnit % 60;
68          }
69          String sTeksti = Integer.toString(sekunnit);
70          String mTeksti = Integer.toString(minuutit);
71          String tTeksti = Integer.toString(tunnit);
72          if (sekunnit < 10) {
73              sTeksti = "0" + sekunnit;
74          }
75          if (minuutit < 10) {
76              mTeksti = "0" + minuutit;
77          }
78          if (tunnit < 10) {
79              tTeksti = "0" + tunnit;
80          }
81          String kAika = tTeksti + ":" + mTeksti + ":" + sTeksti;
82          return kAika;
83      }
84      
85      public void lopetus() throws IOException {
86          aika.stop();
87          int korttiMaara = korttienMaara();
88          int sekunnit = tk.getKokonaisSekunnit();
89          t.paivita(sekunnit, korttiMaara, false);
90          int pelatut = t.getPelatut();
91          int lapaistyt = t.getLapaistyt();
92          int kmKorttiMaara = t.keskimaarainenKorttienMaara();
93          String kmAika = t.keskimaarainenAika();
94          double lp = t.lapaisyProsentti();
95          int valinta = JOptionPane.showConfirmDialog(null, "Ei enää poistettavia kortteja. \n\n"
96                                                             + "Pelatut pelit: " + pelatut + "\n"
97                                                             + "Läpäistyt pelit: " + lapaistyt + " (" + lp + "%)\n"
98                                                             + "Kortteja jäljellä: " + korttienMaara() + "\n"
99                                                             + "Keskimäärin kortteja jäänyt peleissä: " + kmKorttiMaara + "\n"
100                                                            + "Käytit aikaa: " + kaytettyAika() + "\n"
101                                                            + "Keskimääräinen aika peleissä: " + kmAika + "\n"
102                                                            + "\n"
103                                                            + "Aloitetaanko uusi peli?");
104         if (valinta == JOptionPane.YES_OPTION) {
105             try {
106                 kl.reset(kl.getFrame().getContentPane());
107             } catch (IOException ex) {
108                 Logger.getLogger(PaikkaLabelHiirenKuuntelija.class.getName()).log(Level.SEVERE, null, ex);
109             }
110         }
111     }

113     public void lapaisy() throws IOException {
114         aika.stop();
115         int sekunnit = tk.getKokonaisSekunnit();
116         t.paivita(sekunnit, 0, true);
117         int pelatut = t.getPelatut();
118         int lapaistyt = t.getLapaistyt();
119         int kmKorttiMaara = t.keskimaarainenKorttienMaara();
120         String kmAika = t.keskimaarainenAika();
121         double lp = t.lapaisyProsentti();
122         int valinta = JOptionPane.showConfirmDialog(null, "Läpi meni! \n\n"
123                                                            + "Pelatut pelit: " + pelatut + "\n"
124                                                            + "Läpäistyt pelit: " + lapaistyt + " (" + lp + "%)\n"
125                                                            + "Kortteja jäljellä: " + korttienMaara() + "\n"
126                                                            + "Keskimäärin kortteja jäänyt peleissä: " + kmKorttiMaara + "\n"
127                                                            + "Käytit aikaa: " + kaytettyAika() + "\n"
128                                                            + "Keskimääräinen aika peleissä: " + kmAika + "\n"
129                                                            + "\n"
130                                                            + "Aloitetaanko uusi peli?");;
131         if (valinta == JOptionPane.YES_OPTION) {
132             try {
133                 kl.reset(kl.getFrame().getContentPane());
134             } catch (IOException ex) {
135                 Logger.getLogger(PaikkaLabelHiirenKuuntelija.class.getName()).log(Level.SEVERE, null, ex);
136             }
137         }
138     }

140     @Override
141     public void mouseClicked(MouseEvent e) {
142         if (this.vnk.onkoPaalla()) {
143             this.vnk.pois();
144         }
145         if (label.getPaikka().onkoTyhja()) {
146             return;
147         } else if (valitut.isEmpty()) {
148             valitut.add(label);
149             label.getPaikka().asetaValituksi();
150         } else if (valitut.size() == 1) {
151             if (!pt.ovatkoValitutKuvakortteja(label.getPaikka().getKortti(), valitut.get(0).getPaikka().getKortti())) {
152                 if (pt.ovatkoValitutKortitPari(label.getPaikka().getKortti(), valitut.get(0).getPaikka().getKortti()) &&
153                         rt.onkoParinPaikatSamallaRivilla(label.getPaikka(), valitut.get(0).getPaikka())) {
154                     label.getPaikka().tyhjennaPaikka();
155                     valitut.get(0).getPaikka().tyhjennaPaikka();
156                     valitut.get(0).getPaikka().poistaValinta();
157                     valitut.get(0).repaint();
158                     valitut.clear();
159                     tt.lisaaParit();
160                     if (!poyta.onkoKorttienValissaTyhjia() && tt.haeParit().isEmpty()) {
161                         try {
162                             lopetus();
163                         } catch (IOException ex) {
164                             Logger.getLogger(PaikkaLabelHiirenKuuntelija.class.getName()).log(Level.SEVERE, null, ex);
165                         }
166                     }
167                     if (poyta.getTyhjat().size() == 25 && pakka.onkoTyhja()) {
168                         try {
169                             lapaisy();
170                         } catch (IOException ex) {
171                             Logger.getLogger(PaikkaLabelHiirenKuuntelija.class.getName()).log(Level.SEVERE, null, ex);
172                         }
173                     }
174                 } else {
175                     valitut.get(0).getPaikka().poistaValinta();
176                     valitut.get(0).repaint();
177                     valitut.clear();
178                 }
179             } else {
180                 if (rt.onkoParinPaikatSamallaRivilla(label.getPaikka(), valitut.get(0).getPaikka())) {
181                     label.getPaikka().asetaValituksi();
182                     valitut.add(label);
183                 } else {
184                     valitut.get(0).getPaikka().poistaValinta();
185                     valitut.get(0).repaint();
186                     valitut.clear();
187                 }
188             }
189         } else {
190             if (rt.onkoKuvakortitVierekkain(label.getPaikka(), valitut.get(0).getPaikka(), valitut.get(1).getPaikka())) {
191                 label.getPaikka().tyhjennaPaikka();
192                 valitut.get(0).getPaikka().tyhjennaPaikka();
193                 valitut.get(0).getPaikka().poistaValinta();
194                 valitut.get(1).getPaikka().tyhjennaPaikka();
195                 valitut.get(1).getPaikka().poistaValinta();
196                 valitut.get(0).repaint();
197                 valitut.get(1).repaint();
198                 valitut.clear();
199                 tt.lisaaParit();
200                 if (!poyta.onkoKorttienValissaTyhjia() && tt.haeParit().isEmpty()) {
201                     try {
202                         lopetus();
203                     } catch (IOException ex) {
204                         Logger.getLogger(PaikkaLabelHiirenKuuntelija.class.getName()).log(Level.SEVERE, null, ex);
205                     }
206                 }
207                 if (poyta.getTyhjat().size() == 25 && pakka.onkoTyhja()) {
208                     try {
209                         lapaisy();
210                     } catch (IOException ex) {
211                         Logger.getLogger(PaikkaLabelHiirenKuuntelija.class.getName()).log(Level.SEVERE, null, ex);
212                     }
213                 }
214             } else {
215                 valitut.get(0).getPaikka().poistaValinta();
216                 valitut.get(1).getPaikka().poistaValinta();
217                 valitut.get(0).repaint();
218                 valitut.get(1).repaint();
219                 valitut.clear();
220             }
221         }
222         label.repaint();
223     }
