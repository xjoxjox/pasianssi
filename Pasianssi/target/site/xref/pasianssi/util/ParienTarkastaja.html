2   package pasianssi.util;
3   
4   import java.util.HashMap;
5   import pasianssi.domain.Kortti;
6   /**
7    * @author Johanna
8    * Luokka tarkastaa ovatko valitut kortit pareja tai kuvakortteja järjestyksessä.
9    * Parit ovat HashMapissä.
10   */
11  public class ParienTarkastaja {
12      private HashMap<Integer, Integer> parit;
13      /**
14      * Konstruktorissa laitetaan korttiparit HashMap:in.
15      */
16      public ParienTarkastaja() {
17          this.parit = new HashMap<>();
18          this.parit.put(1, 10);
19          this.parit.put(2, 9);
20          this.parit.put(3, 8);
21          this.parit.put(4, 7);
22          this.parit.put(5, 6);
23      }
24      /**
25      * Metodi tarkastaa onko kortti kuvakortti.
26      * @param kortti1 valittu kortti
27      * @see pasianssi.domain.Kortti#getArvo() 
28      * @return kuvakortti
29      */
30      private boolean onkoKuvakortti(Kortti kortti) {
31          return kortti.getArvo() == 11 || kortti.getArvo() == 12 || kortti.getArvo() == 13;
32      }
33      /**
34      * Metodi tarkastaa ovatko valitut kortit kuvakortteja.
35      * @param kortti1 ensimmäiseksi valittu kortti
36      * @param kortti2 toiseksi valittu kortti
37      * @see pasianssi.util.ParienTarkastaja#onkoKuvakortti(Kortti) 
38      * @return kuvakortteja
39      */
40      public boolean ovatkoValitutKuvakortteja(Kortti kortti1, Kortti kortti2) {
41          return onkoKuvakortti(kortti1) && onkoKuvakortti(kortti2);
42      }
43      /**
44      * Metodi tarkastaa ovatko kortit pari.
45      * @param kortti1 ensimmäiseksi valittu kortti
46      * @param kortti2 toiseksi valittu kortti
47      * @see pasianssi.domain.Kortti#getArvo()
48      * @return pari
49      */
50      public boolean ovatkoValitutKortitPari(Kortti kortti1, Kortti kortti2) {
51          if (this.parit.containsKey(kortti1.getArvo())) {
52              if (this.parit.get(kortti1.getArvo()).equals(kortti2.getArvo())) {
53                  return true;
54              }
55          }
56          if (this.parit.containsKey(kortti2.getArvo())) {
57              if (this.parit.get(kortti2.getArvo()).equals(kortti1.getArvo())) {
58                  return true;
59              }
60          }
61          return false;
62      }
63  }
