2   package pasianssi.util;
3   
4   import pasianssi.domain.Paikka;
5   import pasianssi.domain.Pakka;
6   import pasianssi.domain.Poyta;
7   
8   /**
9    * @author Johanna
10   * Luokka jakaa kortit pöydälle hallitsevan tilanteen mukaan.
11   */
12  public class Jakaja {
13      private Pakka pakka;
14      private Poyta poyta;
15      
16      public Jakaja(Pakka pakka, Poyta poyta) {
17          this.pakka = pakka;
18          this.poyta = poyta;
19      }
20      /**
21      * Metodi jakaa kortit pöydän tyhjiin paikkoihin ja jos pakka on tyhjä, 
22      * siirtää käsittelyn metodiin siirraKortitTyhjiin().
23      * @see pasianssi.domain.Poyta#getTyhjat() 
24      * @see pasianssi.domain.Pakka#onkoTyhja()
25      * @see pasianssi.domain.Pakka#poistaKortti(pasianssi.domain.Kortti) 
26      * @see pasianssi.domain.Pakka#pakanKortit() 
27      * @see pasianssi.domain.Paikka#asetaKortti(pasianssi.domain.Kortti) 
28      */
29      public void jaaKortit() {
30          while (true) {
31              if (poyta.getTyhjat().isEmpty()) {
32                  break;
33              }
34              if (this.pakka.onkoTyhja()) {
35                  siirraKortitTyhjiin();
36                  break;
37              }
38              this.poyta.getTyhjat().get(0).asetaKortti(this.pakka.pakanKortit().get(0));
39              this.pakka.poistaKortti(this.pakka.pakanKortit().get(0));
40          }
41      }
42      /**
43      * Metodi siirtää pöydän viimeisenä olevan kortin ensimmäiseen tyhjään paikkaan, 
44      * kunnes korttien välissä ei ole tyhjiä paikkoja.
45      * @see pasianssi.domain.Poyta#onkoKorttienValissaTyhjia() 
46      * @see pasianssi.domain.Poyta#viimeinenPaikkaJossaKortti() 
47      * @see pasianssi.domain.Paikka#asetaKortti(pasianssi.domain.Kortti) 
48      * @see pasianssi.domain.Paikka#tyhjennaPaikka() 
49      */
50      public void siirraKortitTyhjiin() {
51          while (this.poyta.onkoKorttienValissaTyhjia()) {
52              Paikka viimeinen = this.poyta.viimeinenPaikkaJossaKortti();
53              this.poyta.getTyhjat().get(0).asetaKortti(viimeinen.getKortti());
54              viimeinen.tyhjennaPaikka();
55          }
56      }
57  }
